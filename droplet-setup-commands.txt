# Commands to run on your droplet to get the optimized version:

# 1. First, let's see what files you currently have
ls -la

# 2. Check if you have the optimized files
ls -la docker-compose.optimized.yml

# 3. If not, let's pull the latest changes
git pull origin main

# 4. Check again for the optimized files
ls -la docker-compose.optimized.yml

# 5. If still not there, let's check the git status
git status

# 6. If needed, let's force pull the latest
git fetch origin main
git reset --hard origin/main

# 7. Now check for the optimized files
ls -la docker-compose.optimized.yml

# 8. If you have it now, stop current containers
docker-compose down

# 9. Start the optimized version
docker-compose -f docker-compose.optimized.yml up -d

# 10. Check status
docker-compose -f docker-compose.optimized.yml ps
